CREATE TABLE "MetodosAgiles"."Licencia"
(
    "Id_licencia" integer NOT NULL,
    "Clase" text COLLATE pg_catalog."default" NOT NULL,
    "Fecha_emision" date NOT NULL,
    "Fecha_vencimiento" date NOT NULL,
    "Vigencia" integer NOT NULL,
    "Dni_titular" integer NOT NULL,
    "Dni_usuario" integer NOT NULL,
    CONSTRAINT "Licencia_pkey" PRIMARY KEY ("Id_licencia"),
    CONSTRAINT fk_dni_titular FOREIGN KEY ("Dni_titular")
        REFERENCES "MetodosAgiles"."Titular" ("Dni") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT fk_dni_usuario FOREIGN KEY ("Dni_usuario")
        REFERENCES "MetodosAgiles"."Usuario" ("Dni") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;
